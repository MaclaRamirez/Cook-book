\begin{recipe}
[ % Optionale Eingaben
    preparationtime = {\unit[40]{Minuten}},
    bakingtime = {\unit[25]{Minuten}},

        topbottomheat=\unit[200]{\textcelcius}

    portion = {\portion{2}},
    source =  \href{https://www.hellofresh.de/recipes/vegetarisches-massaman-curry-thermomix-5fa27c40bf324869fb598724}{Hello Fresh}
]
{Veganes Gem\"use-Curry}
    \graph
    {% Bilder
        % small=pic/glass,    % kleines Bild
        % big=pic/ingredients % großes (längeres) Bild
    }
    \ingredients
    {% Zutaten
        \unit[1]{} & S\"u{\ss}kartoffel \\
        \unit[2]{} & Kartoffel \\
        \unit[1]{} & Zwiebel \\
        \unit[1]{} & Zucchini \\
        \unit[1]{Dose} & Kichererbsen \\
        \unit[1]{} & Zwiebel \\
        \unit[1]{} & Limette \\
        \unit[1]{} & Chilischote \\
        \unit[1]{} & F\"uhlingszwiebeln  \\
         & Currypaste \\
         & Erdnussbutter \\
         & Kokosmilch \\
         & Gem\"sebr\"uhe \\
    }

    \preparation{%
      \step Backofen auf \unit[200]{\textcelcius}  vorheizen. S\"u{\ss}kartoffel, Kartoffeln und Zucchini in ca. 1 cm gro{\ss}e St\"ucke schneiden. Auf ein Backblech geben, mit Oliven\"ol, Pfeffer und Salz bestreuen und f\"r 25 Minuten in den Backofen geben.

      \step In der Zwischenzeit Kichererbsen abwaschen, die Zwiebeln sch\"alen und in Streifen schneiden. Zwiebeln und Kichererbsen in einer Pfanne anbraten. Ca. 400 ml Gem\"usebr\"uhe aufsetzen.

      \step Die Currypaste zu den Zwiebeln und Kichererbsen hinzugeben und weiter braten. Mit der Gem\"usebr\"uhe abl\"oschen und weiter einkochen lassen.

      \step Kokosmilch und Erdnussbutter hinzugeben und weiter einreduzieren lassen. Mit Pfeffer, Salz und dem Saft einer halben Limitte abschmecken. Das Ofengem\"se in die Currysauce geben und umr\"uhren.

      \step Die Chilischote klein hacken und die Fr\"uhlingszwiebeln klein schneiden. Das Curry auf einem Teller anrichten und mit der kleingehackten Chili und der Fr\"uhlingszwiebel garnieren. Dazu ebenfalls eine Spalte der Limette reichen.
    }

    \hint
    {% Tipp
    }

\end{recipe}
